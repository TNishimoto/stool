221
222 sdsl::int_vector<> _bwt;

226 std::vector<uint64_t> char_counter_vec;

228
229 uint64_t end_marker_position = UINT64_MAX;
230 uint64_t end_marker = UINT64_MAX;
231 for (uint64_t i = 0; i < text_size; i++)
232 {
233 _bwt[i] = bwt[i];
234 char_counter_vec[bwt[i]]++;
235 if((uint64_t)bwt[i] < end_marker){
236 end_marker = bwt[i];
237 end_marker_position = i;

239 }

242 sdsl::wt_gmr<> wt_gmr = stool::bwt::LFDataStructure::construct_wt_gmr(_bwt, stool::Message::increment_paragraph_level(message_paragraph));
243
244 LFDataStructure r;
245 r.bwt.swap(_bwt);
246 r.wt.swap(wt_gmr);
247 r.C.swap(c_array);
248 r.end_marker = end_marker;
249 r.end_marker_position = end_marker_position;

252 st2 = std::chrono::system_clock::now();
253 if (message_paragraph >= 0 && bwt.size() > 0)
