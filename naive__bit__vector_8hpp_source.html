782 template <typename T>
783 void replace_64bit_string_sequence(uint64_t position, const T &bits64_array, uint64_t bit_size, uint64_t array_size)
784 {
785 if (bit_size == 0)
786 {
787 return;
788 }
789
790 uint64_t start_block_index = position / 64;
791 uint64_t start_bit_index = position % 64;
792 uint64_t end_block_index = (position + bit_size - 1) / 64;
793 uint64_t end_bit_index = (position + bit_size - 1) % 64;
794
795 uint64_t blockL_size = start_bit_index;
796 uint64_t blockR_size = 64 - start_bit_index;
797
798 if (start_block_index == end_block_index)
799 {
800 uint64_t old_block = this->buffer_[start_block_index];
801 uint64_t removed_num = stool::Byte::count_bits(old_block);
802 uint64_t new_block = stool::MSBByte::write_bits(old_block, start_bit_index, bit_size, bits64_array[0]);
803 uint64_t added_num = stool::Byte::count_bits(new_block);
804 this->num1_ += added_num;
805 this->num1_ -= removed_num;
806 this->buffer_[start_block_index] = new_block;
807 }
808 else
809 {
810 {
811 uint64_t old_block = this->buffer_[start_block_index];
812 uint64_t removed_num = stool::Byte::count_bits(old_block);
813 uint64_t new_block = stool::MSBByte::write_suffix(old_block, blockR_size, bits64_array[0]);
814 uint64_t added_num = stool::Byte::count_bits(new_block);
815 this->num1_ += added_num;
816 this->num1_ -= removed_num;
817 this->buffer_[start_block_index] = new_block;
818 }
819
820 if (blockL_size == 0)
821 {
822 for (uint64_t i = start_block_index + 1; i < end_block_index; i++)
823 {
824 uint64_t old_block = this->buffer_[i];
825 uint64_t removed_num = stool::Byte::count_bits(old_block);
826 uint64_t new_block = bits64_array[i - start_block_index];
827 uint64_t added_num = stool::Byte::count_bits(new_block);
828 this->num1_ += added_num;
829 this->num1_ -= removed_num;
830 this->buffer_[i] = new_block;
831 }
832 }
833 else
834 {
835 for (uint64_t i = start_block_index + 1; i < end_block_index; i++)
836 {
837 uint64_t old_block = this->buffer_[i];
838 uint64_t removed_num = stool::Byte::count_bits(old_block);
839 uint64_t blockL = bits64_array[(i - 1) - start_block_index] << blockR_size;
840 uint64_t blockR = bits64_array[i - start_block_index] >> blockL_size;
841 uint64_t new_block = blockL | blockR;
842 uint64_t added_num = stool::Byte::count_bits(new_block);
843 this->num1_ += added_num;
844 this->num1_ -= removed_num;
845 this->buffer_[i] = new_block;
846 }
847 }
848
849 {
850 uint64_t last_block_size = end_bit_index + 1;
851 CircularBitPointer start_value_bp(array_size, 0, 0);
852 CircularBitPointer end_value_bp(array_size, 0, 0);
853 start_value_bp.add(bit_size - last_block_size);
854 end_value_bp.add(bit_size - 1);
855 uint64_t pattern = start_value_bp.read64(bits64_array);
856 uint64_t old_block = this->buffer_[end_block_index];
857 uint64_t removed_num = stool::Byte::count_bits(old_block);
858 assert(last_block_size <= 64 && last_block_size > 0);
859 uint64_t new_block = stool::MSBByte::write_prefix(old_block, last_block_size, pattern);
860 uint64_t added_num = stool::Byte::count_bits(new_block);
861 this->num1_ += added_num;
862 this->num1_ -= removed_num;
863 this->buffer_[end_block_index] = new_block;
864 }
