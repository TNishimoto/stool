606 {

613 uint64_t copy_size = old_ending_position - old_starting_position + 1;
614 if (old_starting_position <= old_ending_position)
615 {
616 std::memcpy(tmp_array.data(), &array[old_starting_position], copy_size);
617 }
618 else
619 {

621 uint64_t suffix_size = old_ending_position + 1;
622 std::memcpy(tmp_array.data(), &array[old_starting_position], prefix_size);
623 std::memcpy(&tmp_array[prefix_size], &array[0], suffix_size);
624 }

626 }
627 else if (old_byte_type == ByteType::U16)

629 uint8_t byte_size = 2;
