234 static bool load_bits(std::ifstream &file, CONTAINER &output)
235 {
236 uint64_t xsize = 0;
237 file.read((char *)&xsize, sizeof(uint64_t));
238
239 std::vector<uint8_t> buffer;
240 uint64_t size = (xsize / 8) + ((xsize % 8) == 0 ? 0 : 1);
241 buffer.resize(size, 0);
242 file.read((char *)&buffer[0], sizeof(uint8_t) * buffer.size());
243
244 output.resize(xsize);
245 uint64_t y = 0;
246
247 for (uint64_t i = 0; i < buffer.size(); i++)
248 {
249 uint8_t _byte = buffer[i];
250 uint64_t w = std::min(xsize - (i * 8), (uint64_t)8UL);
251
252 for (size_t j = 0; j < w; ++j)
253 {
254 output[y + w - 1 - j] = _byte & 1;
255 _byte = _byte >> 1;

257 y += w;
258 }

372 static bool write_bits(std::ofstream &out, CONTAINER &text)
373 {
374 if (!out)
375 return 1;
376
377 uint64_t xsize = text.size();
378 out.write((char *)(&xsize), sizeof(uint64_t));
379 uint64_t x = 0;
380 std::vector<uint8_t> buffer;
381 while (x * 8 < text.size())
382 {
383 uint64_t SIZE = text.size() - (x * 8);
384 uint64_t w = std::min(SIZE, (uint64_t)8UL);
385 uint8_t _byte = 0;
386 for (size_t j = 0; j < w; ++j)
387 _byte = (_byte << 1) | text[(x * 8) + j];
388 buffer.push_back(_byte);
389 if (buffer.size() > 8192)
390 {
391 out.write((char *)(&buffer[0]), sizeof(uint8_t) * buffer.size());
392 buffer.clear();
393 }
394 x++;
395 }
396 if (buffer.size() > 0)
397 {
