230 {
231 uint64_t xsize = 0;
232 file.read((char *)&xsize, sizeof(uint64_t));
233
234 std::vector<uint8_t> buffer;
235 uint64_t size = (xsize / 8) + ((xsize % 8) == 0 ? 0 : 1);
236 buffer.resize(size, 0);
237 file.read((char *)&buffer[0], sizeof(uint8_t) * buffer.size());

239 output.resize(xsize);
240 uint64_t y = 0;
241
242 for (uint64_t i = 0; i < buffer.size(); i++)

244 uint8_t _byte = buffer[i];
