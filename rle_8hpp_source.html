120 {
121 if (message_paragraph >= 0 && frle.size() > 0)

123 std::cout << stool::Message::get_paragraph_string(message_paragraph) << "Constructing RLE..." << std::endl;
124 }
125 std::chrono::system_clock::time_point st1, st2;
126 st1 = std::chrono::system_clock::now();
127
128
129 sdsl::int_vector<8> head_char_vec;
130 stool::EliasFanoVector lpos_vec;
131
132 // head_char_vec.width(8);
133 stool::EliasFanoVectorBuilder run_bits;
134 head_char_vec.resize(run_count);
135 run_bits.initialize(frle.size() + 1, run_count + 1);
136 uint64_t currentRunP = 0;
137
138 for (CharacterRun<uint8_t, uint64_t> v : frle)
139 {
140 run_bits.push_bit(true);
141 for (uint64_t i = 1; i <= v.length; i++)
142 {
143 run_bits.push_bit(false);
144 }
145 head_char_vec[currentRunP++] = v.character;
146 }
147 run_bits.push_bit(true);
148 run_bits.finish();
149 lpos_vec.build_from_builder(run_bits);
150
151 RLE<uint8_t> rle;
152 rle.initialize(head_char_vec, lpos_vec, smallest_character);
153
154 st2 = std::chrono::system_clock::now();
155 if (message_paragraph >= 0 && frle.size() > 0)
156 {
157 uint64_t sec_time = std::chrono::duration_cast<std::chrono::seconds>(st2 - st1).count();
158 uint64_t ms_time = std::chrono::duration_cast<std::chrono::milliseconds>(st2 - st1).count();
159 uint64_t per_time = ((double)ms_time / (double)frle.size()) * 1000000;

161 std::cout << stool::Message::get_paragraph_string(message_paragraph) << "[END] Elapsed Time: " << sec_time << " sec (" << per_time << " ms/MB)" << std::endl;
