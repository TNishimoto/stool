124 static std::array<uint64_t, (SIZE * 2) - 1> rebuild_layout(std::array<uint64_t, SIZE> &items)
125 {
126 std::array<uint64_t, (SIZE * 2) - 1> layout;
127
128 uint64_t first_leaf_position = get_first_level_node_position_on_layout(HEIGHT - 1);
129 for(uint64_t i = 0; i < SIZE; i++){
130 layout[first_leaf_position + i] = items[i];
131 }
132 for(int64_t h = (int64_t)HEIGHT - 2; h >= 0; h--){
133 uint64_t first_node_position = get_first_level_node_position_on_layout(h);
134 uint64_t node_count = count_level_nodes(h);
135
136 for(uint64_t i = 0; i < node_count;i++){
137 uint64_t node_position = first_node_position + i;
138 uint64_t left_child_position = ((node_position + 1) * 2) - 1;
139 uint64_t right_child_position = ((node_position + 1) * 2);
140 uint64_t sum = (layout[left_child_position] == DUMMY_VALUE ? 0 : layout[left_child_position]) + (layout[right_child_position] == DUMMY_VALUE ? 0 : layout[right_child_position]);
141 layout[node_position] = sum;
142 }
143 }

582 int64_t search(uint64_t value, uint64_t &sum) const
583 {
584 assert(this->verify());
585
586 uint64_t total = this->psum();
587 uint64_t size = this->size();
588
589 if (value > total || size == 0)
590 {
591 return -1;
592 }
593 else
594 {
595
596 sum = 0;
597 uint64_t pos = 0;
598 uint64_t tmp_value = value;
599 for(int64_t h = 0; h < (int64_t)(HEIGHT - 1); h++){
600 uint64_t left_child_pos = ((pos + 1) * 2) - 1;
601 uint64_t left_value = this->eytzinger_layout_[left_child_pos];
602 bool b = left_value < tmp_value;
603 pos = left_child_pos + b;
604 sum += b * left_value;
605 tmp_value -= b * left_value;
606 }
607 return pos - get_first_level_node_position_on_layout(HEIGHT - 1);
