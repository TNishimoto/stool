75 static uint64_t sum32x2bits(uint64_t B)
76 {
77 uint64_t lsb = B & 0x5555555555555555ULL;
78
79 uint64_t msb = (B >> 1) & 0x5555555555555555ULL;
80
81 uint64_t lsb_sum = stool::Byte::popcount(lsb);
82 uint64_t msb_sum = stool::Byte::popcount(msb);
83
84 return lsb_sum + (2 * msb_sum);
85 }
89 static inline uint64_t sum16x4bits(uint64_t B)
90 {
91 const uint64_t mask = 0x0F0F0F0F0F0F0F0FULL;
92 uint64_t lo = B & mask;
93 uint64_t hi = (B >> 4) & mask;
94 uint64_t bytes = lo + hi;
95
96 return (bytes * 0x0101010101010101ULL) >> 56;
97 }
101 static inline uint32_t sum8x8bits(uint64_t B)
102 {
103 B = (B & 0x00FF00FF00FF00FFULL) + ((B >> 8) & 0x00FF00FF00FF00FFULL);
104 B = (B & 0x0000FFFF0000FFFFULL) + ((B >> 16) & 0x0000FFFF0000FFFFULL);
105 B = (B & 0x00000000FFFFFFFFULL) + (B >> 32);
106 return static_cast<uint32_t>(B); // or (uint32_t)(x & 0x7FF)
107 }

146 static uint64_t psum64x1bits(uint64_t *B, uint64_t i, [[maybe_unused]] uint64_t array_size)
147 {
148 uint64_t block_index = i / 64;
149 uint64_t bit_index = i % 64;
150 uint64_t sum = 0;
151 assert(block_index < array_size);
152
153 for (uint64_t j = 0; j < block_index; j++)
154 {
155 sum += stool::Byte::popcount(B[j]);
156 }
157 uint64_t last_block = B[block_index] >> (63 - bit_index);
158 sum += stool::Byte::popcount(last_block);
159 return sum;
160 }
166 static uint64_t psum64x1bits(uint64_t *B, uint64_t i, uint64_t j, [[maybe_unused]] uint64_t array_size)
167 {
168 uint64_t start_block_index = i / 64;
169 uint64_t start_bit_index = i % 64;
170 uint64_t end_block_index = j / 64;
171 uint64_t end_bit_index = j % 64;
172 uint64_t sum = 0;
173 assert(start_block_index < array_size);
174 assert(end_block_index < array_size);
175
176 if (start_block_index < end_block_index)
177 {
178 uint64_t modified_start_block = B[start_block_index] << start_bit_index;
179 sum += stool::Byte::popcount(modified_start_block);
180
181 for (uint64_t k = start_block_index + 1; k < end_block_index; k++)
182 {
183 sum += stool::Byte::popcount(B[k]);
184 }
185
186 uint64_t modified_last_block = B[end_block_index] >> (63 - end_bit_index);
187 sum += stool::Byte::popcount(modified_last_block);
188 }
189 else
190 {
191 uint64_t maskL = UINT64_MAX >> start_bit_index;
192 uint64_t maskR = UINT64_MAX << (63 - end_bit_index);
193 uint64_t modified_last_block = B[end_block_index] & maskL & maskR;
194 sum += stool::Byte::popcount(modified_last_block);
195 }
196 return sum;
197 }
