103 static uint64_t sum32x2bits(uint64_t bits)
104 {
105 uint64_t lsb = bits & 0x5555555555555555ULL;
106
107 uint64_t msb = (bits >> 1) & 0x5555555555555555ULL;
108
109 uint64_t lsb_sum = stool::Byte::count_bits(lsb);
110 uint64_t msb_sum = stool::Byte::count_bits(msb);
111
112 return lsb_sum + (2 * msb_sum);
113 }
125 static inline uint64_t sum16x4bits(uint64_t x)
126 {
127 const uint64_t mask = 0x0F0F0F0F0F0F0F0FULL;
128 uint64_t lo = x & mask;
129 uint64_t hi = (x >> 4) & mask;
130 uint64_t bytes = lo + hi;
131
132 return (bytes * 0x0101010101010101ULL) >> 56;
133 }

775 static inline uint32_t find_prefix_ge_y_2b32_branchless(uint64_t X, uint32_t y)
776 {
777 const uint64_t M64 = 0x5555555555555555ULL; // ...0101
778 const uint32_t M32 = 0x55555555u;
779 const uint32_t M16 = 0x5555u;
780 const uint32_t M8 = 0x55u;
781 const uint32_t M4 = 0x5u;
782
783 uint32_t total = (uint32_t)__builtin_popcountll(X & M64) + ((uint32_t)__builtin_popcountll((X >> 1) & M64) << 1);
784
785 uint32_t y0 = y;
786 uint32_t idx = 0;
787 uint64_t W = X;
788
789 uint32_t up32 = (uint32_t)(W >> 32);
790 uint32_t sum16 = (uint32_t)__builtin_popcount(up32 & M32) + ((uint32_t)__builtin_popcount((up32 >> 1) & M32) << 1);
791
792 uint32_t c1 = (y > sum16);
793 uint32_t m1 = 0u - c1;
794 idx += (c1 << 4); // +16
795 y -= (sum16 & m1);
796 W >>= ((1u - c1) * 32);
797
798 uint32_t up16 = (uint32_t)((W >> 16) & 0xFFFFu);
799 uint32_t sum8 = (uint32_t)__builtin_popcount(up16 & M16) + ((uint32_t)__builtin_popcount((up16 >> 1) & M16) << 1);
800
801 uint32_t c2 = (y > sum8);
802 uint32_t m2 = 0u - c2;
803 idx += (c2 << 3); // +8
804 y -= (sum8 & m2);
805 W >>= ((1u - c2) * 16);
806
807 uint32_t up8 = (uint32_t)((W >> 8) & 0xFFu);
808 uint32_t sum4 = (uint32_t)__builtin_popcount(up8 & M8) + ((uint32_t)__builtin_popcount((up8 >> 1) & M8) << 1);
809
810 uint32_t c3 = (y > sum4);
811 uint32_t m3 = 0u - c3;
812 idx += (c3 << 2); // +4
813 y -= (sum4 & m3);
814 W >>= ((1u - c3) * 8);
815
816 uint32_t up4 = (uint32_t)((W >> 4) & 0xFu);
817 uint32_t sum2 = (uint32_t)__builtin_popcount(up4 & M4) + ((uint32_t)__builtin_popcount((up4 >> 1) & M4) << 1);
818
819 uint32_t c4 = (y > sum2);
820 uint32_t m4 = 0u - c4;
821 idx += (c4 << 1); // +2
822 y -= (sum2 & m4);
823 W >>= ((1u - c4) * 4);
824
825 uint32_t v_first = (uint32_t)((W >> 2) & 0x3u);
826 uint32_t c5 = (y > v_first);
827 uint32_t p = idx + c5;
828
829 uint32_t cf = (y0 > total);
830 uint32_t mf = 0u - cf;
831 return (p & ~mf) | (32u & mf);
832 }

847 static inline uint32_t find_prefix_ge_y_nib4x16_branchless(uint64_t X, uint32_t y)
848 {
849 const uint64_t NIB = 0x0F0F0F0F0F0F0F0FULL;
850
851 uint64_t bytes = (X & NIB) + ((X >> 4) & NIB); // b0..b7
852 uint64_t s16 = (bytes & 0x00FF00FF00FF00FFULL) + ((bytes >> 8) & 0x00FF00FF00FF00FFULL); // [b0+b1, b2+b3, b4+b5, b6+b7]
853 uint64_t s32 = (s16 & 0x0000FFFF0000FFFFULL) + ((s16 >> 16) & 0x0000FFFF0000FFFFULL); // [b0..b3, b4..b7]
854
855 uint32_t total = (uint32_t)((bytes * 0x0101010101010101ULL) >> 56);
856
857 uint32_t y0 = y; // 番兵判定用に保存
858 uint32_t idx = 0; // 先頭(MSB)から何ニブル進んだか
859
860 uint32_t sum_front8 = (uint32_t)(s32 >> 32); // b4+b5+b6+b7
861 uint32_t c1 = (y > sum_front8); // 0:前半, 1:後半へ
862 uint32_t m1 = 0u - c1; // 0 or 0xFFFFFFFF
863 idx += (c1 << 3); // +8
864 y -= (sum_front8 & m1);
865
866 uint32_t shift16 = 16u + ((1u - c1) << 5); // 16 + (1-c1)*32
867 uint32_t sum_front4 = (uint32_t)((s16 >> shift16) & 0xFFFFu);
868 uint32_t c2 = (y > sum_front4);
869 uint32_t m2 = 0u - c2;
870 idx += (c2 << 2); // +4
871 y -= (sum_front4 & m2);
872
873 uint32_t byte_idx_from_lsb = 7u - (idx >> 1);
874 uint32_t firstByteSum = (uint32_t)((bytes >> (byte_idx_from_lsb * 8u)) & 0xFFu);
875 uint32_t c3 = (y > firstByteSum);
876 uint32_t m3 = 0u - c3;
877 idx += (c3 << 1); // +2
878 y -= (firstByteSum & m3);
879
880 uint32_t nib_shift = (15u - idx) << 2;
881 uint32_t loNib = (uint32_t)((X >> nib_shift) & 0xFu);
882 uint32_t c4 = (y > loNib); // 0/1
883 uint32_t p = idx + c4;
884
885 uint32_t cf = (y0 > total);
886 uint32_t mf = 0u - cf;
887 return (p & ~mf) | (16u & mf);
888 }
902 static inline uint32_t find_prefix_ge_y_nib8x8_branchless(uint64_t X, uint32_t y)
903 {
904 uint64_t s16 = (X & 0x00FF00FF00FF00FFULL) + ((X >> 8) & 0x00FF00FF00FF00FFULL); // [b7+b6, b5+b4, b3+b2, b1+b0] (LSB→MSB)
905
906 uint64_t s32 = (s16 & 0x0000FFFF0000FFFFULL) + ((s16 >> 16) & 0x0000FFFF0000FFFFULL); // low32: b7..b4, high32: b3..b0
907
908 uint32_t sum_back4 = (uint32_t)(s32); // b4..b7
909 uint32_t sum_front4 = (uint32_t)(s32 >> 32); // b0..b3
910 uint32_t total = sum_front4 + sum_back4;
911
912 uint32_t y0 = y;
913 uint32_t idx = 0;
914
915 uint32_t c1 = (y > sum_front4); // 0: 前半, 1: 後半へ
916 uint32_t m1 = 0u - c1; // 0 or 0xFFFFFFFF
917 idx += (c1 << 2); // +4
918 y -= (sum_front4 & m1);
919

922
923 uint32_t c2 = (y > sum_front2); // 0: 先頭ペア, 1: 後ろのペアへ

926 y -= (sum_front2 & m2);
927
