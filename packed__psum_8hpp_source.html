74 static uint64_t sum32x2bits(uint64_t B)
75 {
76 uint64_t lsb = B & 0x5555555555555555ULL;
77
78 uint64_t msb = (B >> 1) & 0x5555555555555555ULL;
79
80 uint64_t lsb_sum = stool::Byte::popcount(lsb);
81 uint64_t msb_sum = stool::Byte::popcount(msb);
82
83 return lsb_sum + (2 * msb_sum);
84 }
88 static inline uint64_t sum16x4bits(uint64_t B)
89 {
90 const uint64_t mask = 0x0F0F0F0F0F0F0F0FULL;
91 uint64_t lo = B & mask;
92 uint64_t hi = (B >> 4) & mask;
93 uint64_t bytes = lo + hi;
94
95 return (bytes * 0x0101010101010101ULL) >> 56;
96 }
100 static inline uint32_t sum8x8bits(uint64_t B)
101 {
102 B = (B & 0x00FF00FF00FF00FFULL) + ((B >> 8) & 0x00FF00FF00FF00FFULL);
103 B = (B & 0x0000FFFF0000FFFFULL) + ((B >> 16) & 0x0000FFFF0000FFFFULL);
104 B = (B & 0x00000000FFFFFFFFULL) + (B >> 32);
105 return static_cast<uint32_t>(B); // or (uint32_t)(x & 0x7FF)
106 }

145 static uint64_t psum64x1bits(uint64_t *B, uint64_t i, [[maybe_unused]] uint64_t array_size)
146 {
147 uint64_t block_index = i / 64;
148 uint64_t bit_index = i % 64;
149 uint64_t sum = 0;
150 assert(block_index < array_size);
151
152 for (uint64_t j = 0; j < block_index; j++)
153 {
154 sum += stool::Byte::popcount(B[j]);
155 }
156 uint64_t last_block = B[block_index] >> (63 - bit_index);
157 sum += stool::Byte::popcount(last_block);
158 return sum;
159 }
165 static uint64_t psum64x1bits(uint64_t *B, uint64_t i, uint64_t j, [[maybe_unused]] uint64_t array_size)
166 {
167 uint64_t start_block_index = i / 64;
168 uint64_t start_bit_index = i % 64;
169 uint64_t end_block_index = j / 64;
170 uint64_t end_bit_index = j % 64;
171 uint64_t sum = 0;
172 assert(start_block_index < array_size);
173 assert(end_block_index < array_size);
174
175 if (start_block_index < end_block_index)
176 {
177 uint64_t modified_start_block = B[start_block_index] << start_bit_index;
178 sum += stool::Byte::popcount(modified_start_block);
179
180 for (uint64_t k = start_block_index + 1; k < end_block_index; k++)
181 {
182 sum += stool::Byte::popcount(B[k]);
183 }
184
185 uint64_t modified_last_block = B[end_block_index] >> (63 - end_bit_index);
186 sum += stool::Byte::popcount(modified_last_block);
187 }
188 else
189 {
190 uint64_t maskL = UINT64_MAX >> start_bit_index;
191 uint64_t maskR = UINT64_MAX << (63 - end_bit_index);
192 uint64_t modified_last_block = B[end_block_index] & maskL & maskR;
193 sum += stool::Byte::popcount(modified_last_block);
194 }
195 return sum;
196 }
