604 void insert(size_t position, const T &value)
605 {
606 if (position > this->size() + 1)
607 {
608 throw std::out_of_range("Position out of range");
609 }
610
611 // 容量が足りない場合は容量を増やす
612
613 if (position == 0)
614 {
615 this->push_front(value);
616 }
617 else if (position == this->size())
618 {
619 this->push_back(value);
620 }
621 else
622 {
623 uint64_t new_byte_type = std::max((uint8_t)get_byte_type(value), this->value_byte_type_);
624 uint64_t new_capacity_size_index = get_appropriate_circular_buffer_size_index(this->deque_size_ + 1);
625 uint64_t old_capacity_size_index = get_appropriate_circular_buffer_size_index(this->circular_buffer_size_ >> (this->value_byte_type_ - 1));
626
627 this->reset_starting_position();
628
629 if (new_byte_type > this->value_byte_type_ || new_capacity_size_index > old_capacity_size_index)
630 {
631 this->shrink_to_fit2(new_capacity_size_index, new_byte_type);
632 }
633 uint64_t value_byte_size = get_byte_size2(this->value_byte_type_);
634
635 uint64_t src_pos = position * value_byte_size;
636 uint64_t dst_pos = src_pos + value_byte_size;
637 uint64_t move_size = this->circular_buffer_size_ - dst_pos;
638
639 memmove(this->circular_buffer_ + dst_pos, this->circular_buffer_ + src_pos, move_size);
640
641 this->deque_size_++;
642 this->set_value(position, value);
643 }
644
645 assert(this->at(position) == value);

709 void shrink_to_fit2(uint64_t capacity_size_index, uint8_t new_byte_type)
710 {
711 uint64_t new_byte_size = get_byte_size2(new_byte_type);
712 uint64_t new_capacity_byte_size = size_array[capacity_size_index] * new_byte_size;
713 uint64_t old_byte_size = get_byte_size2(this->value_byte_type_);
714
715 if (new_capacity_byte_size > ByteArrayDeque<INDEX_TYPE>::max_deque_size())
716 {
717 assert(new_capacity_byte_size > ByteArrayDeque<INDEX_TYPE>::max_deque_size());
718
719 throw std::invalid_argument("shrink_to_fit");
720 }
721 else if (old_byte_size != new_byte_size)
722 {
723 std::array<uint8_t, 65536> tmp_array;
724 uint64_t i = 0;
725 for (uint64_t it : *this)
726 {
727 std::memcpy(&tmp_array[i * new_byte_size], &it, new_byte_size);
728 i++;
729 }
730
731 uint8_t *new_data = new uint8_t[new_capacity_byte_size];
732 std::memcpy(new_data, tmp_array.data(), new_capacity_byte_size);
733
734 if (this->circular_buffer_ != nullptr)
735 {
736 delete[] this->circular_buffer_;
737 this->circular_buffer_ = nullptr;
738 }
739
740 this->circular_buffer_ = new_data;
741 this->starting_position_ = 0;
742 this->circular_buffer_size_ = new_capacity_byte_size;
743 this->value_byte_type_ = new_byte_type;
744 }
745 else if (new_capacity_byte_size > this->circular_buffer_size_ || new_capacity_byte_size < this->circular_buffer_size_)
746 {
747
748 this->reset_starting_position();
749
750 uint8_t *new_data = new uint8_t[new_capacity_byte_size];
751
752 if (new_capacity_byte_size > this->circular_buffer_size_)
753 {
754 std::memcpy(new_data, this->circular_buffer_, this->circular_buffer_size_);
755 }
756 else
757 {
758 std::memcpy(new_data, this->circular_buffer_, new_capacity_byte_size);
759 }
760
761 if (this->circular_buffer_ != nullptr)
762 {
763 delete[] this->circular_buffer_;
764 this->circular_buffer_ = nullptr;
765 }
766
767 this->circular_buffer_ = new_data;
768 this->starting_position_ = 0;
769 this->circular_buffer_size_ = new_capacity_byte_size;
