724 uint8_t code_type_candidate = (uint8_t)stool::PackedPSum::get_code_type(value_v);

737 if (value_v > old_value)

739 this->psum_ += value_v - old_value;

741 else
742 {
743 this->psum_ -= old_value - value_v;

747 this->buffer_[block_index] = stool::MSBByte::write_bits(this->buffer_[block_index], bit_index, code_length, write_value);

854 void push_back(uint64_t value)
855 {
856 uint64_t size = this->size();
857 if (size + 1 > MAX_SIZE)
858 {
859 std::cout << "Error: push_back() " << size + 1 << " > " << MAX_SIZE << std::endl;

864 uint64_t new_code_type = std::max(this->code_type_, code_type_candidate);
865 assert(new_code_type <= 6);
866
867 this->shrink_to_fit(size + 1, new_code_type);

873

877 this->size_++;
