486 void shrink_to_fit(int64_t new_element_count, uint8_t new_code_type)
487 {
488 assert(new_code_type <= 6);
489 uint8_t new_code_length = 1ULL << new_code_type;
490
491 int64_t current_size_index = this->get_current_buffer_size_index();
492 int64_t appropriate_size_index = NaiveFLCVector::get_appropriate_buffer_size_index2(new_element_count, new_code_type);
493
494 if (this->code_type_ == new_code_type)
495 {
496
497 if (appropriate_size_index + 1 < current_size_index || appropriate_size_index > current_size_index)
498 {
499 // this->reset_starting_position();
500 uint64_t old_size = this->buffer_size_;
501 uint64_t *tmp = this->buffer_;
502 assert(tmp != nullptr);
503 assert(appropriate_size_index < (int64_t)size_array.size());
504 uint64_t new_size = size_array[appropriate_size_index];
505 this->buffer_ = new uint64_t[new_size];
506 this->buffer_size_ = new_size;
507
508 // std::array<uint64_t, TMP_BUFFER_SIZE> tmp_array;
509 uint64_t copy_size = std::min(old_size, new_size);
510
511 if (copy_size > 0)
512 {
513 std::memcpy(&this->buffer_[0], &tmp[0], copy_size * sizeof(uint64_t));
514 }
515
516 delete[] tmp;
517 }
518 }
519 else
520 {
521
522 uint64_t *tmp = this->buffer_;
523 assert(tmp != nullptr);
524 assert(appropriate_size_index < (int64_t)size_array.size());
525 uint64_t new_size = size_array[appropriate_size_index];
526 this->buffer_ = new uint64_t[new_size];
527 uint64_t old_code_length = 1ULL << this->code_type_;
528 for (uint64_t i = 0; i < this->size_; i++)
529 {
530 uint64_t old_pos = i << this->code_type_;
531 uint64_t new_pos = i << new_code_type;
532 uint64_t old_block_index = old_pos / 64;
533 uint64_t new_block_index = new_pos / 64;
534 uint64_t old_bit_index = old_pos % 64;
535 uint64_t new_bit_index = new_pos % 64;
536
537 uint64_t old_value = NaiveFLCVector::read_as_64bit_integer(tmp[old_block_index], old_bit_index, old_code_length);
538 old_value = old_value << (64 - new_code_length);
539 this->buffer_[new_block_index] = stool::MSBByte::write_bits(this->buffer_[new_block_index], new_bit_index, new_code_length, old_value);
540 }
541 delete[] tmp;
542
543 this->code_type_ = new_code_type;
544 this->buffer_size_ = new_size;
545 }

555 void push_back(uint64_t value)
556 {
557 uint64_t size = this->size();
558 if (size + 1 > MAX_SIZE)
559 {
560 std::cout << "Error: push_back() " << size + 1 << " > " << MAX_SIZE << std::endl;
561 throw std::invalid_argument("Error: push_back()");
562 }
563
564 uint8_t code_type_candidate = (uint8_t)stool::PackedPSum::get_code_type(value);
565 uint64_t new_code_type = std::max(this->code_type_, code_type_candidate);
566 assert(new_code_type <= 6);
567
568 this->shrink_to_fit(size + 1, new_code_type);
569 uint64_t code_length = 1ULL << new_code_type;
570 uint64_t pos = this->size_ << new_code_type;

572 uint64_t block_index = pos / 64;

574

924 assert(new_code_type <= 6);
