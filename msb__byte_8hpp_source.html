218 static uint64_t fill(uint64_t B, uint64_t pos, uint64_t len, bool b)
219 {
220 uint64_t end_pos = pos + len - 1;
221
222 if (b)
223 {
224 uint64_t mask1 = (UINT64_MAX >> (63 - end_pos)) << (63 - end_pos);
225 uint64_t mask2 = (mask1 << pos) >> pos;
226 return B | mask2;
227 }
228 else
229 {
230 if (pos > 0)
231 {
232 uint64_t pref = UINT64_MAX << (64 - pos);
233
234 if (end_pos >= 63)
235 {
236 uint64_t mask = pref;
237 return B & mask;
238 }
239 else
240 {
241 uint64_t suf = UINT64_MAX >> (end_pos + 1);
242 uint64_t mask = pref | suf;
243 return B & mask;
244 }
245 }
246 else
247 {

250 return 0ULL;
251 }
252 else
253 {

266 static void fill(BIT64_SEQUENCE &B, uint64_t pos, uint64_t len, bool b)
267 {
268 uint64_t current_block_index = pos / 64;
269 uint64_t current_bit_index = pos % 64;
270
271 uint64_t counter = len;
272 while (counter > 0)
273 {
274 if (current_bit_index == 0)
275 {
276 if (counter >= 64)
277 {
278 B[current_block_index] = b ? UINT64_MAX : 0ULL;
279 current_block_index++;
280 counter -= 64;
281 }
282 else
283 {
284
285 B[current_block_index] = fill(B[current_block_index], current_bit_index, counter, b);
286 current_bit_index += counter;
287 counter = 0;
288 }
289 }
290 else
291 {
292 if (current_bit_index + counter >= 64)
293 {
294
295 uint64_t mask_size = 64 - current_bit_index;
296
297 B[current_block_index] = fill(B[current_block_index], current_bit_index, mask_size, b);
298 current_block_index++;
299 current_bit_index = 0;
300 counter -= mask_size;
301 }
302 else
303 {
304
305 B[current_block_index] = fill(B[current_block_index], current_bit_index, counter, b);
306 current_bit_index += counter;
307 counter = 0;
308 }
309 }

476 static void shift_right(uint64_t *B, uint64_t pos, uint64_t len, uint64_t array_size)
477 {
478 uint64_t block_index = pos / 64;
479 uint64_t bit_index = pos % 64;
480
481 uint64_t dst_block_index = (pos + len) / 64;
482 uint64_t dst_bit_index = (pos + len) % 64;
483
484 uint64_t curry_bitsize = 0;
485 uint64_t mid_block_index = 0;
486
487 if (dst_block_index >= array_size)
488 {
489 throw std::runtime_error("shift_rightX: dst_block_index >= array_size");
490 return;
491 }
492 else if (bit_index <= dst_bit_index)
493 {
494 curry_bitsize = dst_bit_index - bit_index;
495 mid_block_index = dst_block_index;
496 // uint64_t shift_block_size = dst_block_index - block_index;
497 uint64_t suffix_block_size = array_size - dst_block_index;
498 std::memmove(&B[dst_block_index], &B[block_index], suffix_block_size * sizeof(uint64_t));
499 }
500 else
501 {
502
503 mid_block_index = dst_block_index - 1;
504 curry_bitsize = dst_bit_index + 64 - bit_index;
505 // uint64_t shift_block_size = (dst_block_index - 1) - block_index;
506 uint64_t suffix_block_size = array_size - (dst_block_index - 1);
507 std::memmove(&B[dst_block_index - 1], &B[block_index], suffix_block_size * sizeof(uint64_t));
508 }
509
510 if (curry_bitsize > 0)
511 {
512 assert(curry_bitsize < 64);
513
514 uint64_t curried_bits = B[mid_block_index];
515 {
516 B[mid_block_index] = stool::MSBByte::shift_right(B[mid_block_index], bit_index, curry_bitsize);
517 }
518
519 for (uint64_t j = mid_block_index + 1; j < array_size; j++)
520 {
521 uint64_t L = curried_bits << (64 - curry_bitsize);
522 uint64_t R = B[j] >> curry_bitsize;
523 curried_bits = B[j];
524 B[j] = L | R;
525 }
526 }
527 fill(B, pos, len, false);

534 static void shift_left(uint64_t *B, uint64_t pos, uint64_t len, uint64_t array_size)
535 {
536 if (len > pos)
537 {
538 throw std::runtime_error("shift_leftX: shift_bitsize > shift_pos");
539 return;
540 }
541
542 uint64_t block_index = pos / 64;
543 uint64_t bit_index = pos % 64;
544
545 uint64_t dst_block_index = (pos - len) / 64;
546 uint64_t dst_bit_index = (pos - len) % 64;
547
548 uint64_t curry_bitsize = 0;
549 uint64_t mid_block_index = 0;
550 uint64_t prefix_block_size = array_size - block_index;
551
552 if (bit_index >= dst_bit_index)
553 {
554 curry_bitsize = bit_index - dst_bit_index;
555 mid_block_index = dst_block_index;
556 // uint64_t shift_block_size = block_index - dst_block_index;
557 uint64_t tmp = B[dst_block_index];
558 std::memmove(&B[dst_block_index], &B[block_index], prefix_block_size * sizeof(uint64_t));
559
560 uint64_t maskL = dst_bit_index > 0 ? (UINT64_MAX << (64 - dst_bit_index)) : 0ULL;
561 uint64_t maskR = UINT64_MAX >> bit_index;
562 B[dst_block_index] = (tmp & maskL) | ((B[dst_block_index] & maskR) << curry_bitsize);
563 }
564 else
565 {
566
567 mid_block_index = dst_block_index + 1;
568 curry_bitsize = bit_index + (64 - dst_bit_index);
569 // uint64_t shift_block_size = block_index - (dst_block_index + 1);
570 std::memmove(&B[dst_block_index + 1], &B[block_index], prefix_block_size * sizeof(uint64_t));
571 }
572
573 if (curry_bitsize > 0)
574 {
575 assert(curry_bitsize < 64);
576 assert(mid_block_index + prefix_block_size - 1 < array_size);
577
578 uint64_t curried_bits = B[mid_block_index + prefix_block_size - 1];
579
580 if (mid_block_index + prefix_block_size - 1 != dst_block_index)
581 {
582 B[mid_block_index + prefix_block_size - 1] = B[mid_block_index + prefix_block_size - 1] << curry_bitsize;
583 }
584
585 for (int64_t j = mid_block_index + prefix_block_size - 2; j >= (int64_t)(dst_block_index + 1); j--)
586 {
587 assert(j < (int64_t)array_size);
588 uint64_t L = B[j] << curry_bitsize;
589 uint64_t R = curried_bits >> (64 - curry_bitsize);
590 curried_bits = B[j];
591 B[j] = L | R;
592 }
593
594 if (bit_index >= dst_bit_index)
595 {
596
597 uint64_t L = B[dst_block_index];
598 uint64_t R = curried_bits >> (64 - curry_bitsize);
599 B[dst_block_index] = L | R;
600 }
601 else
602 {
603 uint64_t Lsize = 64 - dst_bit_index;
604 uint64_t Rsize = curry_bitsize - Lsize;
605 assert(dst_bit_index > 0);
606
607 uint64_t maskL = UINT64_MAX << Lsize;
608 uint64_t maskR = UINT64_MAX << (64 - Lsize);
609
610 uint64_t L = B[dst_block_index] & maskL;
611 uint64_t R = (curried_bits << Rsize) & maskR;
612 B[dst_block_index] = L | (R >> (64 - Lsize));
613 }
614 }
615
616 uint64_t len1 = (array_size * 64) - pos;
617 uint64_t xpos = pos - len + len1;
618 uint64_t xlen = (array_size * 64) - xpos;
619 fill(B, xpos, xlen, false);
620 }
