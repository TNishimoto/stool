216 static uint64_t fill(uint64_t B, uint64_t pos, uint64_t len, bool b)
217 {
218 uint64_t end_pos = pos + len - 1;
219
220 if (b)
221 {
222 uint64_t mask1 = (UINT64_MAX >> (63 - end_pos)) << (63 - end_pos);
223 uint64_t mask2 = (mask1 << pos) >> pos;
224 return B | mask2;
225 }
226 else
227 {
228 if (pos > 0)
229 {
230 uint64_t pref = UINT64_MAX << (64 - pos);
231
232 if (end_pos >= 63)
233 {
234 uint64_t mask = pref;
235 return B & mask;
236 }
237 else
238 {
239 uint64_t suf = UINT64_MAX >> (end_pos + 1);
240 uint64_t mask = pref | suf;
241 return B & mask;
242 }
243 }
244 else
245 {
246 if (end_pos >= 63)
247 {
248 return 0ULL;
249 }
250 else
251 {
252 uint64_t mask = UINT64_MAX >> (end_pos + 1);

264 {
265 uint64_t current_block_index = pos / 64;
266 uint64_t current_bit_index = pos % 64;
267
268 uint64_t counter = len;
269 while (counter > 0)
270 {
271 if (current_bit_index == 0)
272 {
273 if (counter >= 64)
274 {
275 B[current_block_index] = b ? UINT64_MAX : 0ULL;
276 current_block_index++;
277 counter -= 64;
278 }
279 else
280 {

283 current_bit_index += counter;
284 counter = 0;
285 }
286 }

293
294 B[current_block_index] = fill(B[current_block_index], current_bit_index, mask_size, b);

297 counter -= mask_size;

302 B[current_block_index] = fill(B[current_block_index], current_bit_index, counter, b);

565 static void shift_right(uint64_t *B, uint64_t pos, uint64_t len, uint64_t array_size)
566 {
567 uint64_t block_index = pos / 64;
568 uint64_t bit_index = pos % 64;
569
570 uint64_t dst_block_index = (pos + len) / 64;
571 uint64_t dst_bit_index = (pos + len) % 64;
572
573 uint64_t curry_bitsize = 0;
574 uint64_t mid_block_index = 0;
575
576 if (dst_block_index >= array_size)
577 {
578 throw std::runtime_error("shift_rightX: dst_block_index >= array_size");
579 return;
580 }
581 else if (bit_index <= dst_bit_index)
582 {
583 curry_bitsize = dst_bit_index - bit_index;
584 mid_block_index = dst_block_index;
585 // uint64_t shift_block_size = dst_block_index - block_index;
586 uint64_t suffix_block_size = array_size - dst_block_index;
587 std::memmove(&B[dst_block_index], &B[block_index], suffix_block_size * sizeof(uint64_t));
588 }
589 else
590 {
591
592 mid_block_index = dst_block_index - 1;
593 curry_bitsize = dst_bit_index + 64 - bit_index;
594 // uint64_t shift_block_size = (dst_block_index - 1) - block_index;
595 uint64_t suffix_block_size = array_size - (dst_block_index - 1);
596 std::memmove(&B[dst_block_index - 1], &B[block_index], suffix_block_size * sizeof(uint64_t));
597 }
598
599 if (curry_bitsize > 0)
600 {
601 assert(curry_bitsize < 64);
602
603 uint64_t curried_bits = B[mid_block_index];
604 {
605 B[mid_block_index] = stool::MSBByte::shift_right(B[mid_block_index], bit_index, curry_bitsize);
606 }
607
608 for (uint64_t j = mid_block_index + 1; j < array_size; j++)
609 {
610 uint64_t L = curried_bits << (64 - curry_bitsize);
611 uint64_t R = B[j] >> curry_bitsize;
612 curried_bits = B[j];
613 B[j] = L | R;
614 }
615 }
616 fill(B, pos, len, false);

623 static void shift_left(uint64_t *B, uint64_t pos, uint64_t len, uint64_t array_size)
624 {
625 if (len > pos)
626 {
627 throw std::runtime_error("shift_leftX: shift_bitsize > shift_pos");
628 return;
629 }
630
631 uint64_t block_index = pos / 64;
632 uint64_t bit_index = pos % 64;
633
634 uint64_t dst_block_index = (pos - len) / 64;
635 uint64_t dst_bit_index = (pos - len) % 64;
636
637 uint64_t curry_bitsize = 0;
638 uint64_t mid_block_index = 0;
639 uint64_t prefix_block_size = array_size - block_index;
640
641 if (bit_index >= dst_bit_index)
642 {
643 curry_bitsize = bit_index - dst_bit_index;
644 mid_block_index = dst_block_index;
645 // uint64_t shift_block_size = block_index - dst_block_index;
646 uint64_t tmp = B[dst_block_index];
647 std::memmove(&B[dst_block_index], &B[block_index], prefix_block_size * sizeof(uint64_t));
648
649 uint64_t maskL = dst_bit_index > 0 ? (UINT64_MAX << (64 - dst_bit_index)) : 0ULL;
650 uint64_t maskR = UINT64_MAX >> bit_index;
651 B[dst_block_index] = (tmp & maskL) | ((B[dst_block_index] & maskR) << curry_bitsize);
652 }
653 else
654 {
655
656 mid_block_index = dst_block_index + 1;
657 curry_bitsize = bit_index + (64 - dst_bit_index);
658 // uint64_t shift_block_size = block_index - (dst_block_index + 1);
659 std::memmove(&B[dst_block_index + 1], &B[block_index], prefix_block_size * sizeof(uint64_t));
660 }
661
662 if (curry_bitsize > 0)
663 {
664 assert(curry_bitsize < 64);
665 assert(mid_block_index + prefix_block_size - 1 < array_size);
666
667 uint64_t curried_bits = B[mid_block_index + prefix_block_size - 1];
668

672 }
673

675 {
676 assert(j < (int64_t)array_size);
677 uint64_t L = B[j] << curry_bitsize;
678 uint64_t R = curried_bits >> (64 - curry_bitsize);
679 curried_bits = B[j];
680 B[j] = L | R;
681 }
682
683 if (bit_index >= dst_bit_index)
684 {
685
686 uint64_t L = B[dst_block_index];
687 uint64_t R = curried_bits >> (64 - curry_bitsize);
688 B[dst_block_index] = L | R;
689 }
690 else
691 {
692 uint64_t Lsize = 64 - dst_bit_index;
693 uint64_t Rsize = curry_bitsize - Lsize;

697 uint64_t maskR = UINT64_MAX << (64 - Lsize);

699 uint64_t L = B[dst_block_index] & maskL;
700 uint64_t R = (curried_bits << Rsize) & maskR;

703 }

706 uint64_t xpos = pos - len + len1;
707 uint64_t xlen = (array_size * 64) - xpos;
708 fill(B, xpos, xlen, false);
