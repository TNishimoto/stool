217 static int64_t select1(uint64_t bits, uint64_t i)
218 {
219 #if defined(__BMI2__)
220
221
222 unsigned cnt = Byte::count_bits(bits);
223 if (i >= cnt)
224 return -1; // 存在しない
225 uint64_t src = 1ull << i; // r番目の1を表す単一ビット
226 uint64_t bit = _pdep_u64(src, bits); // xの中の該当位置に配置
227 uint64_t _result = static_cast<int>(LSBByte::select1(bit)); // その位置を返す
228
229 return _result;
230
231#else
232
233 // 事前チェック（総1数 < i+1 は失敗）
234 // sの最終バイト = 総1数 なので、本当は後段だけで検出できるが、
235 // 早期終了しておくと無駄を抑えられる（popcountは1命令）。
236 if (i >= static_cast<unsigned>(Byte::count_bits(bits)))
237 return -1;
238
239 // 1) byte popcount（各バイトに 0..8）
240 uint64_t b = bits - ((bits & 0xAAAAAAAAAAAAAAAAULL) >> 1);
241 b = (b & 0x3333333333333333ULL) + ((b >> 2) & 0x3333333333333333ULL);
242 b = (b + (b >> 4)) & 0x0F0F0F0F0F0F0F0FULL;
243
244 // 2) byte prefix sums
245 constexpr uint64_t L8 = 0x0101010101010101ULL;
246 uint64_t s = b * L8; // 各バイト: 先頭からの累積個数
247
248 // 3) 最初に累積 >= i+1 となるバイトを検出
249 constexpr uint64_t H8 = 0x8080808080808080ULL;
250 uint64_t rank = static_cast<uint64_t>(i + 1) * L8;
251 uint64_t mask = ((s | H8) - rank) & H8; // 各バイトのMSBが (s_byte >= rank_byte) なら1
252
253 // 万一該当なし（理論上ここには来ない）は -1
254 if (mask == 0){
255 assert(old_select1(bits, i) == -1);
256 return -1;
257 }
258
259 int byte_index = __builtin_ctzll(mask) / 8;
260
261 // 4) バイト内ランク r とバイト値
262 uint64_t sb = (byte_index == 0) ? 0 : ((s >> ((byte_index - 1) * 8)) & 0xFF);
263 unsigned r = i - static_cast<unsigned>(sb);
264 uint8_t byte = static_cast<uint8_t>(bits >> (byte_index * 8));
265
266 // 5) バイト内 select（テーブル）
267 int bit_in_byte = __LSB_BYTE::select1_table[byte][r];
268 int64_t result = static_cast<int64_t>(byte_index * 8 + bit_in_byte);
269
270 assert(result == old_select1(bits, i));
271 return result;
272#endif
273 }

315 {
