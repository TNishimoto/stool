33 ShortBitVector(const std::vector<bool> &bv)
34 {
35 uint64_t i = 0;
36 while (true)
37 {
38 uint64_t bits = 0;
39 uint64_t start_pos = i * 64;
40 uint64_t end_pos = ((i + 1) * 64) - 1;
41
42 if (end_pos < bv.size())
43 {
44 for (uint64_t x = start_pos; x <= end_pos; x++)
45 {
46 bits = (bits << 1) | (bv[x] ? 1 : 0);
47 }
48 bits_with_gap.push_back(bits);
49 i++;
50 }
51 else
52 {
53 for (uint64_t x = start_pos; x <= end_pos; x++)
54 {
55 if (x < bv.size())
56 {
57 bits = (bits << 1) | (bv[x] ? 1 : 0);
58 }
59 else if (x == bv.size())
60 {
61 bits = (bits << 1) | 1;
62 }
63 else
64 {
65 bits = (bits << 1) | 0;
66 }
67 }
68 bits_with_gap.push_back(bits);
69 break;
70 }
71 }

152 {
153 uint64_t counter = 0;
154 for (uint64_t i = 0; i < this->bits_with_gap.size(); i++)
155 {

157 if (counter + count > nth)

166 uint64_t block_index = (i+1) / 64;
167 uint64_t bit_index = (i+1) % 64;
168 uint64_t current_pos = (i+1);

170

172 uint64_t mask = UINT64_MAX >> bit_index;
173 uint64_t new_bits = bits_with_gap[block_index] & mask;

353 void replace(uint64_t i, uint64_t len, uint64_t value)
354 {
355
356 if (len <= 64)

358 uint64_t block_index = i / 64;

370 this->bits_with_gap[block_index + 1] = stool::MSBByte::write_prefix(this->bits_with_gap[block_index + 1], prefLen, (value << sufLen));
