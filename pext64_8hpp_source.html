34 // 8bit popcount（constexpr）
35 constexpr uint8_t popcount8(uint8_t m)
36 {
37 uint8_t c = 0;
38 while (m)
39 {
40 m = static_cast<uint8_t>(m & (m - 1));
41 ++c;
42 }
43 return c;
44 }

46 // 8bit 圧縮: x のうち mask=1 のビットを LSB から詰める（PEXTの1バイト版）
47 constexpr uint8_t compress_byte(uint8_t x, uint8_t mask)
48 {
49 uint8_t out = 0, k = 0;
50 for (uint8_t b = 0; b < 8; ++b)
51 {
52 if (mask & static_cast<uint8_t>(1u << b))
53 {
54 out |= static_cast<uint8_t>(((x >> b) & 1u) << k);
55 ++k;
56 }
57 }
58 return out;
59 }

61 struct Tables
62 {
63 std::array<uint8_t, 16> pop4{};
64 // comp[mask][byte] = compress_byte(byte, mask)
65 std::array<uint8_t, 256> comp{};
66 };

68 // 64KB のテーブルを constexpr で構築
69 constexpr Tables make_tables()
70 {
71 Tables t{};
72 for (int m = 0; m < 16; ++m)
73 {
74 t.pop4[m] = popcount8(static_cast<uint8_t>(m));
75 for (int x = 0; x < 16; ++x)
76 {
77 t.comp[(m << 4) + x] = compress_byte(static_cast<uint8_t>(x), static_cast<uint8_t>(m));
78 }
79 }
80 return t;
81 }

98 static uint64_t pext_portable(uint64_t x, uint64_t mask)
99 {
100 const auto &tab = pext64::kTables;
101 uint64_t out = 0;
102 unsigned shift = 0;
103 for (unsigned i = 0; i < 16; ++i)
104 {
105 uint8_t mb = (static_cast<uint8_t>(mask >> (i * 4))) & 0b1111;
106 if (!mb)
107 continue;
108 uint8_t xb = (static_cast<uint8_t>(x >> (i * 4))) & 0b1111;
109 uint8_t packed = tab.comp[(mb << 4) + xb];
110 out |= (static_cast<uint64_t>(packed) << shift);
111 shift += tab.pop4[mb];
112 }
113 return out;
114 }

116 static uint64_t naive_pext(uint64_t X, uint64_t Y)
117 {
118 uint64_t out = 0;
119 unsigned i = 0; // out のビット位置（LSBから順に詰める）
120 while (Y)
121 {
122 unsigned b = __builtin_ctzll(Y); // Y の最下位の 1 の位置 = y_i
123 out |= ((X >> b) & 1ull) << i; // X[y_i] を out の i ビット目へ
124 Y &= Y - 1; // 最下位の 1 を落とす
125 ++i;
126 }
127 return out;
128 }

130 static bool has_bmi2_runtime()
131 {
132#if PEXT_RUNTIME_X86
133 unsigned eax = 0, ebx = 0, ecx = 0, edx = 0;
134#if defined(_MSC_VER)
135 int regs[4];
136 __cpuidex(regs, 7, 0);
137 ebx = static_cast<unsigned>(regs[1]);
138#else
139 if (!__get_cpuid_count(7, 0, &eax, &ebx, &ecx, &edx))
140 return false;
141#endif
142 return (ebx & (1u << 8)) != 0; // BMI2
143#else
144 return false;
145#endif
146 }
