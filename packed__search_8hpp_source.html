40 static inline uint32_t find_prefix_ge_y_2b32_branchless(uint64_t B, uint32_t x)
41 {
42 const uint64_t M64 = 0x5555555555555555ULL; // ...0101
43 const uint32_t M32 = 0x55555555u;
44 const uint32_t M16 = 0x5555u;
45 const uint32_t M8 = 0x55u;
46 const uint32_t M4 = 0x5u;
47
48 uint32_t total = (uint32_t)__builtin_popcountll(B & M64) + ((uint32_t)__builtin_popcountll((B >> 1) & M64) << 1);
49
50 uint32_t x0 = x;
51 uint32_t idx = 0;
52 uint64_t W = B;
53
54 uint32_t up32 = (uint32_t)(W >> 32);
55 uint32_t sum16 = (uint32_t)__builtin_popcount(up32 & M32) + ((uint32_t)__builtin_popcount((up32 >> 1) & M32) << 1);
56
57 uint32_t c1 = (x > sum16);
58 uint32_t m1 = 0u - c1;
59 idx += (c1 << 4); // +16
60 x -= (sum16 & m1);
61 W >>= ((1u - c1) * 32);
62
63 uint32_t up16 = (uint32_t)((W >> 16) & 0xFFFFu);
64 uint32_t sum8 = (uint32_t)__builtin_popcount(up16 & M16) + ((uint32_t)__builtin_popcount((up16 >> 1) & M16) << 1);
65
66 uint32_t c2 = (x > sum8);
67 uint32_t m2 = 0u - c2;
68 idx += (c2 << 3); // +8
69 x -= (sum8 & m2);
70 W >>= ((1u - c2) * 16);
71
72 uint32_t up8 = (uint32_t)((W >> 8) & 0xFFu);
73 uint32_t sum4 = (uint32_t)__builtin_popcount(up8 & M8) + ((uint32_t)__builtin_popcount((up8 >> 1) & M8) << 1);
74
75 uint32_t c3 = (x > sum4);
76 uint32_t m3 = 0u - c3;
77 idx += (c3 << 2); // +4
78 x -= (sum4 & m3);
79 W >>= ((1u - c3) * 8);
80
81 uint32_t up4 = (uint32_t)((W >> 4) & 0xFu);
82 uint32_t sum2 = (uint32_t)__builtin_popcount(up4 & M4) + ((uint32_t)__builtin_popcount((up4 >> 1) & M4) << 1);
83
84 uint32_t c4 = (x > sum2);
85 uint32_t m4 = 0u - c4;
86 idx += (c4 << 1); // +2
87 x -= (sum2 & m4);
88 W >>= ((1u - c4) * 4);
89
90 uint32_t v_first = (uint32_t)((W >> 2) & 0x3u);
91 uint32_t c5 = (x > v_first);
92 uint32_t p = idx + c5;
93
94 uint32_t cf = (x0 > total);
95 uint32_t mf = 0u - cf;
96 return (p & ~mf) | (32u & mf);
97 }

104 static inline uint32_t find_prefix_ge_y_nib4x16_branchless(uint64_t B, uint32_t x)
105 {
106 const uint64_t NIB = 0x0F0F0F0F0F0F0F0FULL;
107
108 uint64_t bytes = (B & NIB) + ((B >> 4) & NIB); // b0..b7
109 uint64_t s16 = (bytes & 0x00FF00FF00FF00FFULL) + ((bytes >> 8) & 0x00FF00FF00FF00FFULL); // [b0+b1, b2+b3, b4+b5, b6+b7]
110 uint64_t s32 = (s16 & 0x0000FFFF0000FFFFULL) + ((s16 >> 16) & 0x0000FFFF0000FFFFULL); // [b0..b3, b4..b7]
111
112 uint32_t total = (uint32_t)((bytes * 0x0101010101010101ULL) >> 56);
113
114 uint32_t x0 = x; // 番兵判定用に保存
115 uint32_t idx = 0; // 先頭(MSB)から何ニブル進んだか
116
117 uint32_t sum_front8 = (uint32_t)(s32 >> 32); // b4+b5+b6+b7
118 uint32_t c1 = (x > sum_front8); // 0:前半, 1:後半へ
119 uint32_t m1 = 0u - c1; // 0 or 0xFFFFFFFF
120 idx += (c1 << 3); // +8
121 x -= (sum_front8 & m1);
122
123 uint32_t shift16 = 16u + ((1u - c1) << 5); // 16 + (1-c1)*32
124 uint32_t sum_front4 = (uint32_t)((s16 >> shift16) & 0xFFFFu);
125 uint32_t c2 = (x > sum_front4);
126 uint32_t m2 = 0u - c2;
127 idx += (c2 << 2); // +4
128 x -= (sum_front4 & m2);
129
130 uint32_t byte_idx_from_lsb = 7u - (idx >> 1);
131 uint32_t firstByteSum = (uint32_t)((bytes >> (byte_idx_from_lsb * 8u)) & 0xFFu);
132 uint32_t c3 = (x > firstByteSum);
133 uint32_t m3 = 0u - c3;
134 idx += (c3 << 1); // +2
135 x -= (firstByteSum & m3);
136
137 uint32_t nib_shift = (15u - idx) << 2;
138 uint32_t loNib = (uint32_t)((B >> nib_shift) & 0xFu);
139 uint32_t c4 = (x > loNib); // 0/1
140 uint32_t p = idx + c4;
141
142 uint32_t cf = (x0 > total);
143 uint32_t mf = 0u - cf;
144 return (p & ~mf) | (16u & mf);
145 }
151 static inline uint32_t find_prefix_ge_y_nib8x8_branchless(uint64_t B, uint32_t x)
152 {
153 uint64_t s16 = (B & 0x00FF00FF00FF00FFULL) + ((B >> 8) & 0x00FF00FF00FF00FFULL); // [b7+b6, b5+b4, b3+b2, b1+b0] (LSB→MSB)
154
155 uint64_t s32 = (s16 & 0x0000FFFF0000FFFFULL) + ((s16 >> 16) & 0x0000FFFF0000FFFFULL); // low32: b7..b4, high32: b3..b0
156
157 uint32_t sum_back4 = (uint32_t)(s32); // b4..b7
158 uint32_t sum_front4 = (uint32_t)(s32 >> 32); // b0..b3
159 uint32_t total = sum_front4 + sum_back4;
160
161 uint32_t x0 = x;
162 uint32_t idx = 0;
163
164 uint32_t c1 = (x > sum_front4); // 0: 前半, 1: 後半へ
165 uint32_t m1 = 0u - c1; // 0 or 0xFFFFFFFF
166 idx += (c1 << 2); // +4
167 x -= (sum_front4 & m1);
168
169 uint32_t shift16 = 16u + ((1u - c1) << 5); // c1=0→48, c1=1→16
170 uint32_t sum_front2 = (uint32_t)((s16 >> shift16) & 0xFFFFu);
171
172 uint32_t c2 = (x > sum_front2); // 0: 先頭ペア, 1: 後ろのペアへ
173 uint32_t m2 = 0u - c2;
174 idx += (c2 << 1); // +2
175 x -= (sum_front2 & m2);
176
177 uint32_t byte_idx_from_lsb = 7u - idx; // LSB基準インデックス
178 uint32_t firstByte = (uint32_t)((B >> (byte_idx_from_lsb * 8u)) & 0xFFu);
179
180 uint32_t c3 = (x > firstByte); // 0: そこで到達, 1: 次の1要素へ
181 uint32_t p = idx + c3;
182
183 uint32_t cf = (x0 > total);
184 uint32_t mf = 0u - cf;
185 return (p & ~mf) | (8u & mf);
186 }

193 static inline uint32_t find_prefix_ge_y_nib16x4_branchless(uint64_t B, uint32_t x)
194 {
195 uint32_t e0 = (uint32_t)((B >> 48) & 0xFFFFu);
196 uint32_t e1 = (uint32_t)((B >> 32) & 0xFFFFu);
197 uint32_t e2 = (uint32_t)((B >> 16) & 0xFFFFu);
198 uint32_t e3 = (uint32_t)(B & 0xFFFFu);
199
200 uint32_t s0 = e0;
201 uint32_t s1 = s0 + e1;
202 uint32_t s2 = s1 + e2;
203 uint32_t s3 = s2 + e3;
204
205 uint32_t m = ((s0 >= x) ? 1u : 0u) | ((s1 >= x) ? 2u : 0u) | ((s2 >= x) ? 4u : 0u) | ((s3 >= x) ? 8u : 0u);
206
207 m |= ((m == 0u) ? 16u : 0u);
208
209 return (uint32_t)__builtin_ctz(m);
210 }
216 static int64_t search64x1bits(uint64_t *B, uint64_t x, [[maybe_unused]] uint64_t array_size)
217 {
218 uint64_t sum = 0;
219 if (x == 0)
220 {
221 return 0;
222 }
223 sum = 0;
224 uint64_t k = 0;
225 uint64_t v = stool::Byte::popcount(B[k]);
226
227 while (sum + v < x)
228 {
229 sum += v;
230 k++;
231 assert(k < array_size);
232 v = stool::Byte::popcount(B[k]);
233 }
234
235 uint64_t diff = x - sum;
236 uint64_t f = stool::MSBByte::select1(B[k], diff - 1);
237 sum = x - 1;
238 return (k * 64) + f;
239 }

420 static int64_t search(uint64_t *B, uint64_t x, PackedBitType bit_type, uint64_t total_sum, [[maybe_unused]] uint64_t array_size)
421 {
422 if (x > total_sum)
423 {
424 return -1;
425 }
426 switch (bit_type)
427 {
428 case PackedBitType::BIT_1:
429 return search64x1bits(B, x, array_size);
430 case PackedBitType::BIT_2:
431 return search32x2bits(B, x, array_size);
432 case PackedBitType::BIT_4:
433 return search16x4bits(B, x, array_size);
434 case PackedBitType::BIT_8:
435 return search8x8bits(B, x, array_size);
436 case PackedBitType::BIT_16:
437 return search4x16bits(B, x, array_size);
438 case PackedBitType::BIT_32:
439 return search2x32bits(B, x, array_size);
440 case PackedBitType::BIT_64:
441 return search1x64bits(B, x, array_size);
442 default:
443 break;
444 }
445 throw std::invalid_argument("Invalid bit type");
446 }
