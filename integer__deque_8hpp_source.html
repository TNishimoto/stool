55 {

70 static uint64_t set_code(uint64_t code, uint8_t pos, uint8_t byte_size, uint64_t value)
71 {
72 uint64_t left_size = (8 * byte_size) * pos;
73 uint64_t right_size = 64 - (left_size + (8 * byte_size));
74

76 uint64_t right_code = (code << (left_size + (8 * byte_size))) >> (left_size + (8 * byte_size));

695 void shrink_to_fit(uint64_t capacity_bit_size, uint8_t byte_size)
696 {
697 uint64_t size = 1 << capacity_bit_size;
698 uint64_t tsize = size / (8 / byte_size);
699
700 if (tsize > IntegerDeque<INDEX_TYPE>::max_deque_size())
701 {
702 assert(tsize > IntegerDeque<INDEX_TYPE>::max_deque_size());
703
704 throw std::invalid_argument("shrink_to_fit");
705 }
706 else if (size > this->deque_size_ || this->value_byte_size_ != byte_size)
707 {
708 std::cout << "SHRINK/" << tsize << "/" << (uint64_t)byte_size << std::endl;
709
710 T *new_data = new T[tsize];
711 uint64_t i = 0;
712 uint64_t shift = 64 - (byte_size * 8);
713 for (uint64_t i = 0; i < tsize; i++)
714 {
715 new_data[i] = 0;
716 }
717 for (IntegerDequeIterator it = this->begin(); it != this->end(); ++it)
718 {
719 new_data[i] = new_data[i] | (*it << shift);
720 if (shift == 0)
721 {
722 i++;
723 shift = 64 - (byte_size * 8);
724 }
725 else
726 {
727 shift -= byte_size * 8;
728 }
729 }
730
731 if (this->circular_buffer_ != nullptr)
732 {
733 delete[] this->circular_buffer_;
734 this->circular_buffer_ = nullptr;
735 }
736
737 this->circular_buffer_ = new_data;
738 this->starting_position_ = 0;
739 this->circular_buffer_size_ = tsize;
740 this->value_byte_size_ = byte_size;
