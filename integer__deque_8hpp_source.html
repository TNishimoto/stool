53 static uint64_t access_value(uint64_t code, uint8_t pos, uint8_t byte_size)
54 {
55 uint64_t left_size = (8 * byte_size) * pos;

69 static uint64_t set_code(uint64_t code, uint8_t pos, uint8_t byte_size, uint64_t value)
70 {
71 uint64_t left_size = (8 * byte_size) * pos;
72 uint64_t right_size = 64 - (left_size + (8 * byte_size));
73
74 uint64_t left_code = code >> (right_size + (8 * byte_size));
75 uint64_t right_code = (code << (left_size + (8 * byte_size))) >> (left_size + (8 * byte_size));
76
77 return (left_code << (right_size + (8 * byte_size))) | (value << right_size) | right_code;

598 void insert(size_t position, const T &value)
599 {
600 if (position > this->size() + 1)
601 {
602 throw std::out_of_range("Position out of range");
603 }
604
605 // 容量が足りない場合は容量を増やす
606
607 if (position == 0)
608 {
609 this->push_front(value);
610 }
611 else if (position == this->size())
612 {
613 this->push_back(value);
614 }
615 else
616 {
617 uint64_t vsize = stool::LSBByte::get_code_length(value);
618 if (vsize > this->value_byte_size_ * 8)
619 {
620 uint64_t new_byte_size = get_byte_size(value);
621 this->reserve(this->get_buffer_bit(), new_byte_size);
622 }
623
624 this->update_size_if_needed();
625
626
627 for (size_t i = this->deque_size_; i > position; --i)
628 {
629 uint64_t pos = this->starting_position_ + i;
630 if (pos >= this->circular_buffer_size_)
631 {
632 pos -= this->circular_buffer_size_;
633 }
634 this->circular_buffer_[pos] = pos > 0 ? this->circular_buffer_[pos - 1] : this->circular_buffer_[this->circular_buffer_size_ - 1];
635 }
636
637 uint64_t wpos = this->starting_position_ + position;
638 if (wpos >= this->circular_buffer_size_)
639 {
640 wpos -= this->circular_buffer_size_;
641 }
642 this->circular_buffer_[wpos] = value;
643
644 this->deque_size_++;
645 }
646
647 assert(this->at(position) == value);
648 }

694 void shrink_to_fit(uint64_t capacity_bit_size, uint8_t byte_size)
695 {
696 uint64_t size = 1 << capacity_bit_size;
697 uint64_t tsize = size / (8 / byte_size);
698
699 if (tsize > IntegerDeque<INDEX_TYPE>::max_deque_size())
700 {
701 assert(tsize > IntegerDeque<INDEX_TYPE>::max_deque_size());
702
703 throw std::invalid_argument("shrink_to_fit");
704 }
705 else if (size > this->deque_size_ || this->value_byte_size_ != byte_size)
706 {
707 std::cout << "SHRINK/" << tsize << "/" << (uint64_t)byte_size << std::endl;
708
709 T *new_data = new T[tsize];
710 uint64_t i = 0;
711 uint64_t shift = 64 - (byte_size * 8);
712 for (uint64_t i = 0; i < tsize; i++)
713 {
714 new_data[i] = 0;
715 }
716 for (IntegerDequeIterator it = this->begin(); it != this->end(); ++it)
717 {
718 new_data[i] = new_data[i] | (*it << shift);
719 if (shift == 0)
720 {
721 i++;
722 shift = 64 - (byte_size * 8);
723 }
724 else
725 {
726 shift -= byte_size * 8;
727 }
728 }
729
730 if (this->circular_buffer_ != nullptr)
731 {
732 delete[] this->circular_buffer_;
733 this->circular_buffer_ = nullptr;
734 }
735
736 this->circular_buffer_ = new_data;
737 this->starting_position_ = 0;
738 this->circular_buffer_size_ = tsize;
739 this->value_byte_size_ = byte_size;
740 }
