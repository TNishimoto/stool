29 struct lower_bound_unrolled_range
30 {
31 static std::size_t eval(const uint64_t *a, uint64_t v)
32 {
33 if constexpr (L == R){
34 return L;
35 }
36 else if constexpr (L + 1 == R)
37 {
38 // lower_bound: a[L] >= v ? L : R
39 return (a[L] < v) ? R : L;
40 }
41 else
42 {
43 constexpr std::size_t M = (L + R) / 2;
44
45 if (a[M] < v)
46 {
47 return lower_bound_unrolled_range<M + 1, R>::eval(a, v);
48 }
49 else
50 {
51 return lower_bound_unrolled_range<L, M>::eval(a, v);
