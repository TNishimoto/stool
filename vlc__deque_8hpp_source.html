481 static uint64_t access_value(uint64_t code1, uint64_t code2, uint8_t pos, uint8_t len)
482 {
483 if (len > 0)
484 {
485 uint64_t end_pos = pos + len - 1;
486 if (end_pos <= 63)
487 {
488 uint64_t right_gap = 63 - end_pos;
489 uint64_t left_gap = pos;
490 return (code1 << left_gap) >> (left_gap + right_gap);
491 }
492 else if (pos <= 63 && end_pos <= 127)
493 {
494 uint64_t left_code = (code1 << pos) >> pos;
495 uint64_t right_code = code2 >> (127 - end_pos);
496 return (left_code << (end_pos - 63)) | right_code;
497 }
498 else if (pos > 63 && end_pos <= 127)
499 {
500 uint64_t pos1 = pos - 64;
501 uint64_t end_pos1 = end_pos - 64;
502 uint64_t right_gap = 63 - end_pos1;
503 uint64_t left_gap = pos1;
504 return (code2 << left_gap) >> (left_gap + right_gap);
505 }
506 else
507 {
508 throw std::invalid_argument("Error: VLCDeque::access_value");
509 }
510 }
511 else
512 {
513 return 0;
514 }

589 {

706 void push_back(uint64_t v)
707 {
708
709 uint64_t value_length = stool::LSBByte::get_code_length(v);
710 this->value_length_deque.push_back(value_length);
711 if (v > 0)
712 {
713 uint64_t last_code = this->code_deque[this->code_deque.size() - 1];
714
715 if (value_length <= this->last_gap)
716 {
717 this->code_deque[this->code_deque.size() - 1] = last_code | (v << (this->last_gap - value_length));
718 this->last_gap -= value_length;
719 }
720 else
721 {
722 uint64_t len1 = value_length - this->last_gap;
723 uint64_t new_last_gap = 64 - len1;
724
725 uint64_t new_left_code = last_code | (v >> len1);
726 uint64_t new_right_code = v << new_last_gap;
727
728 this->code_deque[this->code_deque.size() - 1] = new_left_code;
729 this->code_deque.push_back(new_right_code);
730 assert(this->code_deque[this->code_deque.size() - 2] == new_left_code);
731
732 this->last_gap = new_last_gap;
733 }
734
735 if (this->last_gap == 0)
736 {
737 this->code_deque.push_back(0);
738 this->last_gap = 64;
739 }
740 }
741
742 assert(this->verify());
743 }
