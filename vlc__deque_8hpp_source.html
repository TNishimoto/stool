496 static uint64_t access_value(uint64_t code1, uint64_t code2, uint8_t pos, uint8_t len)
497 {
498 if (len > 0)
499 {
500 uint64_t end_pos = pos + len - 1;
501 if (end_pos <= 63)
502 {
503 uint64_t right_gap = 63 - end_pos;
504 uint64_t left_gap = pos;
505 return (code1 << left_gap) >> (left_gap + right_gap);
506 }
507 else if (pos <= 63 && end_pos <= 127)
508 {
509 uint64_t left_code = (code1 << pos) >> pos;
510 uint64_t right_code = code2 >> (127 - end_pos);
511 return (left_code << (end_pos - 63)) | right_code;
512 }
513 else if (pos > 63 && end_pos <= 127)
514 {
515 uint64_t pos1 = pos - 64;
516 uint64_t end_pos1 = end_pos - 64;
517 uint64_t right_gap = 63 - end_pos1;
518 uint64_t left_gap = pos1;
519 return (code2 << left_gap) >> (left_gap + right_gap);
520 }
521 else
522 {
523 throw std::invalid_argument("Error: VLCDeque::access_value");
524 }
525 }
526 else
527 {
528 return 0;
529 }

603 static std::pair<uint64_t, uint64_t> insert_zero_bits(uint64_t code, uint8_t pos, uint8_t len)
604 {
605 if (len == 0)
606 {
607 return std::pair<uint64_t, uint64_t>(code, 0);

610 uint64_t rLen = 64 - pos;
611 uint64_t sgap = std::min(rLen, (uint64_t)len);

614 uint64_t _right_tmp = code << pos;

616
617 uint64_t val1 = _left | _right;

721 void push_back(uint64_t v)
722 {
723
724 uint64_t value_length = stool::LSBByte::get_code_length(v);
725 this->value_length_deque.push_back(value_length);
726 if (v > 0)
727 {
728 uint64_t last_code = this->code_deque[this->code_deque.size() - 1];
729
730 if (value_length <= this->last_gap)
731 {
732 this->code_deque[this->code_deque.size() - 1] = last_code | (v << (this->last_gap - value_length));
733 this->last_gap -= value_length;
734 }
735 else
736 {
737 uint64_t len1 = value_length - this->last_gap;
738 uint64_t new_last_gap = 64 - len1;
739
740 uint64_t new_left_code = last_code | (v >> len1);
741 uint64_t new_right_code = v << new_last_gap;
742
743 this->code_deque[this->code_deque.size() - 1] = new_left_code;
744 this->code_deque.push_back(new_right_code);
745 assert(this->code_deque[this->code_deque.size() - 2] == new_left_code);
746
747 this->last_gap = new_last_gap;
748 }
749
750 if (this->last_gap == 0)
751 {
752 this->code_deque.push_back(0);
753 this->last_gap = 64;
754 }
755 }
756
757 assert(this->verify());
758 }

1218 void shift_left(uint64_t pos)
1219 {
1220 if (pos > 0 && pos < this->size())
1221 {
1222 uint64_t value_len = this->value_length_deque[pos];
1223 if (value_len > 0)
1224 {
1225 std::pair<uint64_t, uint8_t> code_pos = this->get_code_starting_position(pos);
1226 if (code_pos.first + 1 < this->code_deque.size())
1227 {
1228 std::pair<uint64_t, uint64_t> rmv_result = VLCDeque::renove_range(this->code_deque[code_pos.first], this->code_deque[code_pos.first + 1], code_pos.second, value_len);
1229 this->code_deque[code_pos.first] = rmv_result.first;
1230 this->code_deque[code_pos.first + 1] = rmv_result.second;
1231 }
1232 else
1233 {
1234 std::pair<uint64_t, uint64_t> rmv_result = VLCDeque::renove_range(this->code_deque[code_pos.first], 0, code_pos.second, value_len);
1235 this->code_deque[code_pos.first] = rmv_result.first;
1236 }
1237 for (size_t i = code_pos.first + 2; i < this->code_deque.size(); i++)
1238 {
1239 uint64_t left = this->code_deque[i] >> (64 - value_len);
1240 uint64_t right = value_len < 64 ? (this->code_deque[i] << value_len) : 0;
1241 this->code_deque[i - 1] = this->code_deque[i - 1] | left;
1242 this->code_deque[i] = right;
1243 }
1244
1245 this->value_length_deque.erase(this->value_length_deque.begin() + pos);
1246
1247 if (this->last_gap + value_len <= 64)
1248 {
1249 this->last_gap += value_len;
1250 }
1251 else
1252 {
1253 if (this->size() > 0)
1254 {
1255 this->code_deque.pop_back();
1256 this->last_gap = ((uint64_t)this->last_gap + (uint64_t)value_len) - 64;
1257 }
1258 else
1259 {
1260 this->clear();
1261 }
1262 }
1263 }
1264 else
1265 {
1266
1267 this->value_length_deque.erase(this->value_length_deque.begin() + pos);
1268 }
1269 }
1270 else if (pos == 0)
1271 {
1272 this->pop_front();
1273 }
1274 else if (pos == this->size())
1275 {
1276 this->pop_back();
1277 }
1278 else

1291 void shift_right(uint64_t pos, uint64_t value_len)
1292 {
1293 if (pos > 0 && pos < this->size())
1294 {
1295 if (value_len > 0)
1296 {
1297 int64_t epos = this->code_deque.size() - 1;
1298 this->code_deque.push_back(0);
1299 std::pair<uint64_t, uint8_t> code_pos = this->get_code_starting_position(pos);
1300
1301 for (int64_t i = epos; i > (int64_t)code_pos.first; --i)
1302 {
1303 std::pair<uint64_t, uint64_t> ins_result = VLCDeque::insert_zero_bits(this->code_deque[i], 0, value_len);
1304 this->code_deque[i + 1] = this->code_deque[i + 1] | ins_result.second;
1305 this->code_deque[i] = ins_result.first;
1306 }
1307
1308 // this->code_deque[i+1] = this->code_deque[i+1] | ((this->code_deque[i] << (64 - value_len)) >> (64 - value_len));
1309
1310 std::pair<uint64_t, uint64_t> ins_result = VLCDeque::insert_zero_bits(this->code_deque[code_pos.first], code_pos.second, value_len);
1311
1312 this->code_deque[code_pos.first] = ins_result.first;
1313 this->code_deque[code_pos.first + 1] = ins_result.second | this->code_deque[code_pos.first + 1];
1314
1315 this->value_length_deque.insert(this->value_length_deque.begin() + pos, value_len);
1316
1317 if (value_len < this->last_gap)
1318 {
1319 this->code_deque.pop_back();
1320 this->last_gap -= value_len;
1321 }
1322 else if (value_len == this->last_gap)
1323 {
1324 this->last_gap = 64;
1325 }
1326 else
1327 {
1328 this->last_gap = 64 - (value_len - this->last_gap);
1329 }
1330 }
1331 else
1332 {
1333 this->value_length_deque.insert(this->value_length_deque.begin() + pos, 0);
1334 }
1335 }
1336 else if (pos == 0)
1337 {
1338 if (value_len > 0)
1339 {
1340 this->push_front(1 << (value_len - 1));
1341 }
1342 else
1343 {
1344 this->push_front(0);
1345 }
1346 }
1347 else if (pos == this->size())
1348 {
1349 if (value_len > 0)
1350 {
1351 this->push_back(1 << (value_len - 1));
1352 }
1353 else
1354 {
1355 this->push_back(0);
1356 }
1357 }
1358 else
1359 {
1360 throw std::invalid_argument("Error: VLCDeque::shift_right()");
1361 }
