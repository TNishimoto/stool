469 static uint64_t access_value(uint64_t code1, uint64_t code2, uint8_t pos, uint8_t len)
470 {
471 if (len > 0)
472 {
473 uint64_t end_pos = pos + len - 1;
474 if (end_pos <= 63)
475 {
476 uint64_t right_gap = 63 - end_pos;
477 uint64_t left_gap = pos;
478 return (code1 << left_gap) >> (left_gap + right_gap);
479 }
480 else if (pos <= 63 && end_pos <= 127)
481 {
482 uint64_t left_code = (code1 << pos) >> pos;
483 uint64_t right_code = code2 >> (127 - end_pos);
484 return (left_code << (end_pos - 63)) | right_code;
485 }
486 else if (pos > 63 && end_pos <= 127)
487 {
488 uint64_t pos1 = pos - 64;
489 uint64_t end_pos1 = end_pos - 64;
490 uint64_t right_gap = 63 - end_pos1;
491 uint64_t left_gap = pos1;
492 return (code2 << left_gap) >> (left_gap + right_gap);
493 }
494 else
495 {
496 throw std::invalid_argument("Error: VLCDeque::access_value");
497 }
498 }
499 else
500 {
501 return 0;
502 }

576 static std::pair<uint64_t, uint64_t> insert_zero_bits(uint64_t code, uint8_t pos, uint8_t len)
577 {
578 if (len == 0)
579 {
580 return std::pair<uint64_t, uint64_t>(code, 0);
581 }
582
583 uint64_t rLen = 64 - pos;
584 uint64_t sgap = std::min(rLen, (uint64_t)len);
585
586 uint64_t _left = stool::Byte::zero_pad_tail(code, 64 - pos);
587 uint64_t _right_tmp = code << pos;
588 uint64_t _right = pos + sgap < 64 ? (_right_tmp >> (pos + sgap)) : 0;
589
590 uint64_t val1 = _left | _right;

592 uint64_t end_pos = pos + len - 1;
593 uint64_t val2 = 0;
594 if (end_pos < 64)

602 val2 = result_pref_length < 64 ? _right_tmp >> result_pref_length : 0;

694 void push_back(uint64_t v)
695 {
696
697 uint64_t value_length = stool::Byte::get_code_length(v);
698 this->value_length_deque.push_back(value_length);
699 if (v > 0)
700 {
701 uint64_t last_code = this->code_deque[this->code_deque.size() - 1];
702
703 if (value_length <= this->last_gap)
704 {
705 this->code_deque[this->code_deque.size() - 1] = last_code | (v << (this->last_gap - value_length));
706 this->last_gap -= value_length;
707 }
708 else
709 {
710 uint64_t len1 = value_length - this->last_gap;
711 uint64_t new_last_gap = 64 - len1;
712
713 uint64_t new_left_code = last_code | (v >> len1);
714 uint64_t new_right_code = v << new_last_gap;
715
716 this->code_deque[this->code_deque.size() - 1] = new_left_code;
717 this->code_deque.push_back(new_right_code);
718 assert(this->code_deque[this->code_deque.size() - 2] == new_left_code);
719
720 this->last_gap = new_last_gap;
721 }
722
723 if (this->last_gap == 0)
724 {
725 this->code_deque.push_back(0);
726 this->last_gap = 64;
727 }
728 }
729
730 assert(this->verify());
731 }

1194 {

1218 this->value_length_deque.erase(this->value_length_deque.begin() + pos);

1264 void shift_right(uint64_t pos, uint64_t value_len)
1265 {
