579 void insert(size_t position, const T &value)
580 {
581 if (this->size() >= SimpleDeque<T, INDEX_TYPE>::max_deque_size())
582 {
583 throw std::invalid_argument("Error: push_back()");
584 }
585
586 if (position > this->size() + 1)
587 {
588 throw std::out_of_range("Position out of range");
589 }
590
591 // 容量が足りない場合は容量を増やす
592
593 if (position == 0)
594 {
595 this->push_front(value);
596 }
597 else if (position == this->size())
598 {
599 this->push_back(value);
600 }
601 else
602 {
603 this->update_size_if_needed();
604
605 // 後ろの要素を1つずつシフトして空間を作る
606 for (size_t i = this->deque_size_; i > position; --i)
607 {
608 uint64_t pos = this->starting_position_ + i;
609 if (pos >= this->circular_buffer_size_)
610 {
611 pos -= this->circular_buffer_size_;
612 }
613 this->circular_buffer_[pos] = pos > 0 ? this->circular_buffer_[pos - 1] : this->circular_buffer_[this->circular_buffer_size_ - 1];
614 }
615
616 uint64_t wpos = this->starting_position_ + position;
617 if (wpos >= this->circular_buffer_size_)
618 {
619 wpos -= this->circular_buffer_size_;
620 }
621 this->circular_buffer_[wpos] = value;
622
623 this->deque_size_++;
624 }
625
626 assert(this->at(position) == value);
627 }
