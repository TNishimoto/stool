586 void insert(size_t position, const T &value)
587 {
588 if (this->size() >= SimpleDeque<T, INDEX_TYPE>::max_deque_size())
589 {
590 throw std::invalid_argument("Error: push_back()");
591 }
592
593 if (position > this->size() + 1)
594 {
595 throw std::out_of_range("Position out of range");
596 }
597
598 // 容量が足りない場合は容量を増やす
599
600 if (position == 0)
601 {
602 this->push_front(value);
603 }
604 else if (position == this->size())
605 {
606 this->push_back(value);
607 }
608 else
609 {
610 this->update_size_if_needed();
611
612 // 後ろの要素を1つずつシフトして空間を作る
613 for (size_t i = this->deque_size_; i > position; --i)
614 {
615 uint64_t pos = this->starting_position_ + i;
616 if (pos >= this->circular_buffer_size_)
617 {
618 pos -= this->circular_buffer_size_;
619 }
620 this->circular_buffer_[pos] = pos > 0 ? this->circular_buffer_[pos - 1] : this->circular_buffer_[this->circular_buffer_size_ - 1];
621 }
622
623 uint64_t wpos = this->starting_position_ + position;
624 if (wpos >= this->circular_buffer_size_)
625 {
626 wpos -= this->circular_buffer_size_;
627 }
628 this->circular_buffer_[wpos] = value;
629
630 this->deque_size_++;
631 }
632
633 assert(this->at(position) == value);
634 }
