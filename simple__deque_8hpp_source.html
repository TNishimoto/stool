588 void insert(size_t position, const T &value)
589 {
590 if (this->size() >= SimpleDeque<T, INDEX_TYPE>::max_deque_size())
591 {
592 throw std::invalid_argument("Error: push_back()");
593 }
594
595 if (position > this->size() + 1)
596 {
597 throw std::out_of_range("Position out of range");
598 }
599
600 // 容量が足りない場合は容量を増やす
601
602 if (position == 0)
603 {
604 this->push_front(value);
605 }
606 else if (position == this->size())
607 {
608 this->push_back(value);
609 }
610 else
611 {
612 this->update_size_if_needed();
613
614 // 後ろの要素を1つずつシフトして空間を作る
615 for (size_t i = this->deque_size_; i > position; --i)
616 {
617 uint64_t pos = this->starting_position_ + i;
618 if (pos >= this->circular_buffer_size_)
619 {
620 pos -= this->circular_buffer_size_;
621 }
622 this->circular_buffer_[pos] = pos > 0 ? this->circular_buffer_[pos - 1] : this->circular_buffer_[this->circular_buffer_size_ - 1];
623 }
624
625 uint64_t wpos = this->starting_position_ + position;
626 if (wpos >= this->circular_buffer_size_)
627 {
628 wpos -= this->circular_buffer_size_;
629 }
630 this->circular_buffer_[wpos] = value;
631
632 this->deque_size_++;
633 }
